int locateElementByBinarySearch(int temp) {
    int low = 0;
    int high = 4;
    int local = 0;
    while(low < high) {
        local = (low+high)/2;
        if(temp > c[local]) {
            low = local + 1;
        } else {
            high = local;
        }
    }
    return low;
}
